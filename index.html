from scipy import stats

# Define the period of interest
period_start_date = '2003-12-01'
period_end_date = '2004-02-29' # Include February 2004

# Filter data for the specific period
period_data = data[(data['date'] >= period_start_date) & (data['date'] <= period_end_date)].copy()

# Calculate the average transfers during the period
average_transfers_period = period_data['Tranfers'].mean()

# Calculate the overall average transfers (excluding the period of interest to avoid bias in comparison)
# Let's re-calculate overall mean excluding the period of interest for a cleaner comparison
data_excluding_period = data[(data['date'] < period_start_date) | (data['date'] > period_end_date)].copy()
overall_average_transfers_excluded = data_excluding_period['Tranfers'].mean()


print(f"Average transfers during Dec 2003 - Feb 2004: {average_transfers_period:.3f}")
print(f"Overall average transfers (excluding period): {overall_average_transfers_excluded:.3f}")

# Perform a one-sample t-test to compare the mean of the period data to the overall mean
# Null hypothesis: The mean of transfers in the period is equal to the overall mean
# Alternative hypothesis: The mean of transfers in the period is greater than the overall mean (one-sided test)
# We need to use the individual data points from the period for the t-test

period_transfers = period_data['Tranfers']

# Perform a one-sample t-test
# We are testing if the mean of the sample (period_transfers) is significantly different from the population mean (overall_average_transfers_excluded)
# For a one-sided test (greater than), we can look at the t-statistic and p-value

t_statistic, p_value = stats.ttest_1samp(period_transfers, overall_average_transfers_excluded, alternative='greater')

print(f"\nOne-Sample T-Test Results:")
print(f"T-statistic: {t_statistic:.3f}")
print(f"P-value: {p_value:.3f}")

# Interpret the results
alpha = 0.05
if p_value < alpha:
    print(f"\nSince the p-value ({p_value:.3f}) is less than the significance level ({alpha}), we reject the null hypothesis.")
    print("This suggests that the average number of transfers during Dec 2003 - Feb 2004 was statistically significantly higher than the overall average transfers.")
else:
    print(f"\nSince the p-value ({p_value:.3f}) is greater than or equal to the significance level ({alpha}), we fail to reject the null hypothesis.")
    print("This suggests that there is not enough statistical evidence to conclude that the average number of transfers during Dec 2003 - Feb 2004 was statistically significantly higher than the overall average transfers.")
